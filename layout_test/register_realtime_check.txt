<?php
session_start();

$login_success = null;

// AUTO LOGIN nếu có cookie (remember)
if (!isset($_SESSION['idkhachhang']) && isset($_COOKIE['remember_email']) && isset($_COOKIE['remember_pass'])) {
    $cookie_email = mysqli_real_escape_string($mysqli, $_COOKIE['remember_email']);
    $cookie_pass  = mysqli_real_escape_string($mysqli, $_COOKIE['remember_pass']);
    $q = mysqli_query($mysqli, "SELECT * FROM tbl_dangky WHERE email = '$cookie_email' AND matkhau = '$cookie_pass' AND trangthai = 1 LIMIT 1");
    if ($q && mysqli_num_rows($q) > 0) {
        $r = mysqli_fetch_array($q);
        $_SESSION['dangky']      = $r['tenkhachhang'];
        $_SESSION['email']       = $r['email'];
        $_SESSION['idkhachhang'] = $r['iddangky'];
        header("Location: index.php");
        exit;
    }
}

// XỬ LÝ LOGIN
if (isset($_POST['dangNhap'])) {
    $email          = mysqli_real_escape_string($mysqli, trim($_POST['email']));
    $plain_password = $_POST['matkhau'];
    $matkhau        = md5($plain_password);
    $remember       = isset($_POST['remember']) ? true : false;

    $sql   = "SELECT * FROM tbl_dangky WHERE email = '$email' AND matkhau = '$matkhau' AND trangthai = 1 LIMIT 1";
    $query = mysqli_query($mysqli, $sql);

    if ($query && mysqli_num_rows($query) > 0) {
        $row = mysqli_fetch_array($query);
        $_SESSION['dangky']      = $row['tenkhachhang'];
        $_SESSION['email']       = $row['email'];
        $_SESSION['idkhachhang'] = $row['iddangky'];

        if ($remember) {
            setcookie("remember_email", $email, time() + 60 * 60 * 24 * 30, "/");
            setcookie("remember_pass", $matkhau, time() + 60 * 60 * 24 * 30, "/");
        } else {
            setcookie("remember_email", "", time() - 3600, "/");
            setcookie("remember_pass", "", time() - 3600, "/");
        }

        header("Location: index.php");
        exit;
    } else {
        $login_success = false;
    }
}

<script>
function closeModal() {
  document.getElementById('loginModal').style.display = 'none';
}
function togglePassword(inputId, eyeId) {
  const passwordInput = document.getElementById(inputId);
  const eyeIcon = document.getElementById(eyeId);

  if (passwordInput.type === "password") {
    passwordInput.type = "text";
    eyeIcon.setAttribute("name", "eye-off-outline");
  } else {
    passwordInput.type = "password";
    eyeIcon.setAttribute("name", "eye-outline");
  }
}
function checkPassword(value) {
  document.getElementById("rule-length").style.color = value.length >= 6 ? "green" : "red";
  document.getElementById("rule-length").textContent = (value.length >= 6 ? "✓" : "✗") + " Ít nhất 6 ký tự";

  const hasNumber = /\d/.test(value);
  document.getElementById("rule-number").style.color = hasNumber ? "green" : "red";
  document.getElementById("rule-number").textContent = (hasNumber ? "✓" : "✗") + " Có ít nhất 1 số";

  const hasUpper = /[A-Z]/.test(value);
  document.getElementById("rule-upper").style.color = hasUpper ? "green" : "red";
  document.getElementById("rule-upper").textContent = (hasUpper ? "✓" : "✗") + " Có ít nhất 1 chữ hoa";

  const hasLower = /[a-z]/.test(value);
  document.getElementById("rule-lower").style.color = hasLower ? "green" : "red";
  document.getElementById("rule-lower").textContent = (hasLower ? "✓" : "✗") + " Có ít nhất 1 chữ thường";
}
